第一部分：入门篇

第一章 游戏概述与设计理念

1.1 什么是复古文字冒险RPG？

复古文字冒险RPG是一种结合了文字冒险游戏和角色扮演游戏元素的游戏类型。它继承了早期计算机游戏的风格，主要通过文字描述来呈现游戏世界，玩家通过选择不同的选项来推进游戏进程。

核心特点：

· 文字驱动：游戏世界通过文字描述呈现
· 回合制战斗：经典的RPG战斗系统
· 角色成长：等级、经验、属性提升
· 物品收集：丰富的道具系统
· 任务系统：主线与支线任务
· 成就系统：记录玩家的特殊成就

1.2 设计目标

本游戏的设计目标包括：

1. 易上手：简单的界面和操作
2. 内容丰富：多样化的场景、敌人、物品
3. 可扩展：便于添加新内容
4. 跨平台：基于Python，可在多平台运行
5. 学习友好：代码结构清晰，适合学习

1.3 技术选型

组件 技术 说明
GUI框架 Tkinter Python内置，无需额外安装
图像处理 PIL/Pillow 处理游戏图标和图片
数据存储 JSON/文本 存档格式
并发处理 threading 处理耗时操作

6.2 属性系统设计

基础属性：

· 生命值(HP)：角色承受伤害的能力
· 攻击力(Attack)：造成伤害的能力
· 防御力(Defense)：减少所受伤害的能力
· 经验值(EXP)：积累升级
· 等级(Level)：影响基础属性

计算公式：

```
实际伤害 = max(1, 攻击力 - 敌人防御力 + 随机浮动)
升级所需经验 = 当前等级 × 100
```

6.3 经验值与升级系统

```python
def gain_exp(self, amount):
    """获得经验值"""
    self.exp += amount
    
    while self.exp >= self.exp_to_next_level():
        self.level_up()

def exp_to_next_level(self):
    """计算升级所需经验值"""
    return self.level * 100

def level_up(self):
```

```python
    """计算魔法伤害"""
    config = self.magic_config[self.magic_affinity]
    base_damage = config['base_damage']
    
    damage = base_damage + self.magic_power
    damage *= (1 + (self.magic_level - 1) * 0.1)
    
    # 属性克制
    if self.magic_affinity == "light" and enemy_type == "dark":
        damage *= 1.5
```

6.5 背包系统

```python
def add_item(self, item_name, quantity=1):
    """添加物品到背包"""
    if item_name in self.inventory:
        self.inventory[item_name] += quantity
    else:
        self.inventory[item_name] = quantity

def remove_item(self, item_name, quantity=1):
    """从背包移除物品"""
    if item_name in self.inventory:
        self.inventory[item_name] -= quantity
        if self.inventory[item_name] <= 0:
            del self.inventory[item_name]

def use_item(self, item_name):
    """使用物品"""
    if item_name in self.inventory and self.inventory[item_name] > 0:
        self.remove_item(item_name, 1)
        return True
    return False
```

8.3 战斗核心逻辑

```python
def start_battle(self, enemy_name):
    """开始战斗"""
    enemy_data = self.enemies[enemy_name]
    diff_settings = self.difficulty_settings[self.config['difficulty']]
    
    # 计算敌人属性（考虑难度和等级缩放）
    enemy_hp = int(enemy_data['hp'] * diff_settings['monster_hp_multiplier'])
    enemy_attack = int(enemy_data['attack'] * diff_settings['monster_attack_multiplier'])
    enemy_defense = int(enemy_data['defense'] * diff_settings['monster_defense_multiplier'])
    
    self.add_message(f"⚔️ 你遇到了 {enemy_name}！", 'warning')
    
    battle_round = 1
    original_hp = enemy_hp
    
    while enemy_hp > 0 and self.player.hp > 0:
        self.add_message(f"--- 回合 {battle_round} ---", 'info')
        
        # 玩家攻击
        damage = self.calculate_player_damage(enemy_defense)
        enemy_hp -= damage
        self.add_message(f"你对 {enemy_name} 造成了 {damage} 点伤害！", 'success')
        
        if enemy_hp <= 0:
            self.handle_victory(enemy_name, enemy_data, diff_settings)
            break
        
        # 敌人攻击
        damage = self.calculate_enemy_damage(enemy_attack)
        self.player.hp = max(0, self.player.hp - damage)
        self.add_message(f"{enemy_name} 对你造成了 {damage} 点伤害！", 'error')
        
        if self.player.hp <= 0:
            self.handle_defeat()
            break
        
        battle_round += 1
        time.sleep(1)  # 战斗动画延迟
```

8.4 伤害计算

```python
def calculate_player_damage(self, enemy_defense):
    """计算玩家造成的伤害"""
    base_damage = max(1, self.player.attack - enemy_defense)
    damage_variation = random.randint(-2, 3)
    return max(1, base_damage + damage_variation)

def calculate_enemy_damage(self, enemy_attack):
    """计算敌人造成的伤害"""
    base_damage = max(1, enemy_attack - self.player.defense)
    damage_variation = random.randint(-2, 3)
    return max(1, base_damage + damage_variation)
```

8.5 战斗奖励处理

```python
def handle_victory(self, enemy_name, enemy_data, diff_settings):
    """处理战斗胜利"""
    exp_gained = int(enemy_data['exp'] * diff_settings['exp_multiplier'])
    gold_gained = int(enemy_data['gold'] * diff_settings['gold_multiplier'])
    
    self.player.gain_exp(exp_gained)
    self.player.gold += gold_gained
    
    self.add_message(f"战斗胜利！获得 {exp_gained} 经验值和 {gold_gained} 金币", 'success')
    
    # 掉落物品
    if 'drops' in enemy_data and enemy_data['drops']:
        for drop_item in enemy_data['drops']:
            if random.random() < diff_settings['item_drop_chance']:
                self.player.add_item(drop_item, 1)
                self.add_message(f"{enemy_name} 掉落了 {drop_item}！", 'success')
```

```python
    if item_name not in self.player.inventory:

    if self.player.equipped[slot]:
```

```python
def load_save_game(self, save_file):
    """从文件加载游戏"""
    try:
        save_path = os.path.join(self.saves_dir, save_file)
        
        with open(save_path, 'r') as f:
            lines = f.readlines()
        
        save_data = {
            "player": {},
            "game_state": {}
        }
        
        current_section = None
        
        for line in lines:
            line = line.strip()
            if not line:
                continue
            
            if line.startswith("==="):
                if "PLAYER DATA" in line:
                    current_section = "player"
                elif "INVENTORY" in line:
                    current_section = "inventory"
                    save_data["player"]["inventory"] = {}
                elif "GAME STATE" in line:
                    current_section = "game_state"
                continue
            
            if current_section == "player":
                if ":" in line:
                    key, value = line.split(":", 1)
                    if key in ["LEVEL", "EXP", "HP", "MAX_HP", "ATTACK", "DEFENSE", "GOLD"]:
                        save_data["player"][key.lower()] = int(value.strip())
                    else:
                        save_data["player"][key.lower()] = value.strip()
            
            elif current_section == "inventory":
                if ":" in line:
                    item, quantity = line.split(":", 1)
                    save_data["player"]["inventory"][item.strip()] = int(quantity.strip())
            
            elif current_section == "game_state":
                if ":" in line:
                    key, value = line.split(":", 1)
                    if key == "DAY_COUNT":
                        save_data["game_state"][key.lower()] = int(value.strip())
                    else:
                        save_data["game_state"][key.lower()] = value.strip()
        
        # 重建玩家对象
        self.player = Player(
            save_data["player"]["name"],
            save_data["player"]["max_hp"],
            save_data["player"]["attack"],
            save_data["player"]["defense"]
        )
        self.player.level = save_data["player"]["level"]
        self.player.exp = save_data["player"]["exp"]
        self.player.hp = save_data["player"]["hp"]
        self.player.gold = save_data["player"]["gold"]
        self.player.inventory = save_data["player"]["inventory"]
        
        # 恢复游戏状态
        self.current_scene = save_data["game_state"]["current_scene"]
        self.game_time = datetime.datetime.fromisoformat(save_data["game_state"]["game_time"])
        self.day_count = save_data["game_state"]["day_count"]
        
        return True
        
    except Exception as e:
        print(f"加载游戏失败: {e}")
        return False
```

```python
def get_save_files(self):

    saves = []
    
    if not os.path.exists(self.saves_dir):
        os.makedirs(self.saves_dir)
    
    for file in os.listdir(self.saves_dir):
        if file.startswith("save_"):
            file_path = os.path.join(self.saves_dir, file)
            try:
                with open(file_path, 'r') as f:
                    # 读取存档信息
                    for line in f:
                        if line.startswith("TIMESTAMP:"):
                            timestamp = line.split(":", 1)[1].strip()
                        elif line.startswith("NAME:"):
                            player_name = line.split(":", 1)[1].strip()
                            break
                
                saves.append({
                    "file": file,
                    "name": player_name,
                    "date": timestamp
                })
            except:
                continue
    
    return saves
```

```python
def scale_enemy_by_level(self, enemy_data, player_level):
    """根据玩家等级缩放敌人属性"""
    level_scaling = (player_level - 1) * 0.1
    
    scaled_hp = int(enemy_data['hp'] * (1 + level_scaling))
    scaled_attack = int(enemy_data['attack'] * (1 + level_scaling))
    scaled_defense = int(enemy_data['defense'] * (1 + level_scaling))
    
    # 确保不会超过上限
    scaled_hp = min(scaled_hp, enemy_data['hp'] * 5)
    scaled_attack = min(scaled_attack, enemy_data['attack'] * 3)
    scaled_defense = min(scaled_defense, enemy_data['defense'] * 3)
    
    return scaled_hp, scaled_attack, scaled_defense
```

```python
    results = {
        'easy': {'wins': 0, 'losses': 0, 'avg_damage': 0},
        'normal': {'wins': 0, 'losses': 0, 'avg_damage': 0},
        'hard': {'wins': 0, 'losses': 0, 'avg_damage': 0}
    }
    
    for difficulty in results.keys():
        self.config['difficulty'] = difficulty
        
        for _ in range(100):  # 测试100次战斗
            result = self.simulate_battle("野狼")
            if result['win']:
                results[difficulty]['wins'] += 1
            else:
                results[difficulty]['losses'] += 1
            results[difficulty]['avg_damage'] += result['damage_taken']
        
        results[difficulty]['avg_damage'] /= 100
    
    return results
```

```python
import time

def measure_performance(func):
    """性能监控装饰器"""
    def wrapper(*args, **kwargs):
        start_time = time.time()
        result = func(*args, **kwargs)
        end_time = time.time()
        
        duration = (end_time - start_time) * 1000  # 转换为毫秒
        if duration > 100:  # 超过100ms的操作为慢操作
            print(f"⚠️ 慢操作: {func.__name__} 耗时 {duration:.2f}ms")
        
        return result
    return wrapper

# 使用示例
@measure_performance
def load_large_data(self):
    """加载大量数据"""
    pass
```
